# 一、SVG

  • SVG意为可缩放矢量图形（Scalable Vector Graphics），使用 XML 格式定义图像，是目前最最火热的图像文件格式之一。

  • SVG其实是一种开放标准的矢量图形语言，可让你设计激动人心的、高分辨率的Web图形页面。用户可以直接用代码来描绘图像，可以用任何文字处理工具打开SVG图像，通过改变部分代码来使图像具有交互功能，并可以随时插入到HTML中通过浏览器来观看。

  • 最重要特性：SVG 图像在放大或改变尺寸的情况下其图形质量不会有所损失（因为是矢量图形）

  • 平均来讲，SVG文件比JPEG和PNG格式的文件要小很多

  • SVG图形是可交互的和动态的，可以在SVG文件中嵌入动画元素或通过脚本来定义动画。

  • SVG提供了3种类型的图形对象：矢量图形（vectorgraphicshape例如：由直线和曲线组成的路径）、图象(image)、文本(text)。图形对象还可进行分组、添加样式、变换、组合等操作，特征集包括嵌套变换（nestedtransformations）、剪切路径（clippingpaths）、alpha蒙板（alphamasks）、滤镜效果（filtereffects）、模板对象（templateobjects）和其它扩展（extensibility）

# 二、SVG.js

SVG.js是一款操作 SVG元素 和创建 SVG 动画的轻量的js库。

原生js创建一个svg图形是这样的：

var ns = 'http://www.w3.org/2000/svg'
var div = document.getElementById('drawing') 
var svg = document.createElementNS(ns, 'svg')
svg.setAttributeNS(null, 'width', '100%')
svg.setAttributeNS(null, 'height', '100%')
div.appendChild(svg)
var rect = document.createElementNS(ns, 'rect')
rect.setAttributeNS(null, 'width', 100)
rect.setAttributeNS(null, 'height', 100)
rect.setAttributeNS(null, 'fill', '##f06')
svg.appendChild(rect)

使用svg.js库后：

var draw = SVG('drawing')
var rect = draw.rect(100, 100).fill('##f06')

语法符合人类习惯，易读。 流行svg基础库大小对比： svg.js在压缩后大小约16k。

# 2. focus实现

import "svg.js"
function FocusHandler(el) {
  this.el = el
  el.remember('_focusHandler', this)
}
FocusHandler.prototype.init = function (value) {
  this.el.first().stroke(value == true ? "green" : "gray").attr({ 'stroke-width': value == true ? 3 : 1 })
}

SVG.extend(SVG.Element, {
  focus: function (value) {
    var selectHandler = this.remember('_focusHandler') || new FocusHandler(this)
    selectHandler.init(value === undefined ? true : value)

    return this
  }
})

上述代码利用SVG.extend扩展了Element类的方法focus，实现时，把第一个元素的画笔颜色和画笔宽度改变了，从而实现获得焦点时的特殊效果

# 3. 自定义类型

import "svg.js"
SVG.NodeHttp = SVG.invent({
  // Define the type of element that should be created
  create: 'g'

  // Specify from which existing class this shape inherits
  , inherit: SVG.G

  // Add custom methods to invented shape
  , extend: {
    genMember: function () {
      return {
        l: 0,
        t: 0,
        dockerPos: { x: 6, y: 30 },
        interfacesPos: [{ x: 95, y: 20 }, { x: 95, y: 40 }],
        linksFrom: [],
        linksTo: [],
      }
    },
    build: function (desc) {
      this.data = this.genMember()
      this.rect(100, 60).fill('none').stroke('gray').radius(5).id('outline')
      this.rect(8, 8).move(2, 26).fill('none').stroke('blue').radius(2).id('docker')
      this.text(desc).id('desc').center(50, 30)
      this.circle(10).fill('blue').move(90, 15).id('int1').attr({ class: 'interface' })
      this.circle(10).fill('blue').move(90, 35).id('int2').attr({ class: 'interface' })
      return this
    },
    desc: function (description) {
      this.select('text').first().text(description).center(50, 30)
      return this
    },
    inside: function (x, y) {
      x = x - this.data.l
      y = y - this.data.t
      let box = this.bbox()

      return x > box.x
        && y > box.y
        && x < box.x + box.width
        && y < box.y + box.height
    },
    interface: function (x, y) {
      for (let i of this.data.interfacesPos) {
        if (Math.pow(x - (i.x + this.data.l), 2) + Math.pow(y - (i.y + this.data.t), 2) <= 100) {
          return { x: i.x + this.data.l, y: i.y + this.data.t }
        }
      }
      return null
    },
    move: function (x = 0, y = 0, box = this.bbox()) {
      this.data.l += x
      this.data.t += y
      const dx = x - box.x
      const dy = y - box.y
      this.dmove(dx, dy)
      this.notifyLink(x, y)
      return this
    },
    docker: function () {
      return { x: this.data.dockerPos.x + this.data.l, y: this.data.dockerPos.y + this.data.t }
    },
    addLinkFrom: function (link) {
      this.data.linksFrom.push(link)
    },
    addLinkto: function (link) {
      this.data.linksTo.push(link)
    },
    notifyLink(x, y) {
      let l
      for (l of this.data.linksFrom) {
        l.fromMoved(x, y)
      }

      for (l of this.data.linksTo) {
        l.toMoved(x, y)
      }
    }
  }

  // Add method to parent elements
  , construct: {
    // Create a http element
    http: function (description) {
      return this.put(new SVG.NodeHttp).build(description)
    }
  }
})

上述代码继承svg.js的G类（对应svg中的group），扩展了父类的方法http（用于实例化NodeHttp类），创建了NodeHttp类，，自定义了相关方法 在自定义的build方法中，生成具体的子元素。通过NodeHttp创建的元素效果：一个group，里面自动创建了很多rect、text、circle等元素。创建方式

    let http3 = this.draw.http("node3").move(130, 110)

    this.allNodes.push(http1)
    this.allNodes.push(http2)

# 6. 图形之间连接

扩展类型Link:

import "svg.js"
SVG.Link = SVG.invent({
  create: 'path'
  , inherit: SVG.Path
  , extend: {
    from: null,
    to: null,
    dockerPos: { x: 6, y: 30 },
    interfacesPos: [{ x: 95, y: 20 }, { x: 95, y: 40 }],
    build: function (from) {
      this.from = from
      return this.stroke('red').attr({ 'stroke-width': 2 })
    },
    inside: function (x, y) {
      x = x - this.l
      y = y - this.t
      let box = this.bbox()

      return x > box.x
        && y > box.y
        && x < box.x + box.width
        && y < box.y + box.height
    },
    lineTo: function (to) {
      this.to = to
      let p = 'M ' + this.from.x + ' ' + this.from.y + ' L ' + this.to.x + ' ' + this.to.y
      return this.plot(p)
    },
    fromMoved: function (x, y) {
      this.from.x += x
      this.from.y += y
      this.lineTo(this.to)
    },
    toMoved: function (x, y) {
      this.to.x += x
      this.to.y += y
      this.lineTo(this.to)
    }
  }

  , construct: {
    link: function (from) {
      return this.put(new SVG.Link).build(from)
    }
  }
})

更新鼠标事件：

import "svg.js"
import "@/lib/svg/svg.focus.js"
import "@/lib/svg/svg.NodeHttp.js"
import "@/lib/svg/svg.Link.js"

export default class AdapterManager {

  constructor(draw) {
    this.draw = draw

    this.allNodes = []
    this.allLinks = []
    this.rootNode = null
    this.currentNode = null
    this.mousedownPos = null
    this.drawLinker = null

    let that = this
    draw.on("mousedown", function (e) {
      that.onMouseDown.call(that, e)
    })
    draw.on("mousemove", function (e) {
      that.onMouseMove.call(that, e)
    })
    draw.on("mouseup", function (e) {
      that.onMouseUp.call(that, e)
    })
  }

  init() {
    let http1 = this.draw.http("node1").move(10, 10)
    let http2 = this.draw.http("node2").move(130, 30)
    let http3 = this.draw.http("node3").move(130, 110)

    this.allNodes.push(http1)
    this.allNodes.push(http2)
    this.allNodes.push(http3)

    this.rootNode = http1
  }

  onMouseDown(e) {
    if (e.button != 0) {
      return
    }
    let node = this.getMouseNode(e.offsetX, e.offsetY)
    if (!node || node != this.currentNode) {
      return
    }

    let intf = node.interface(e.offsetX, e.offsetY)
    if (!intf) {
      this.mousedownPos = { x: e.offsetX, y: e.offsetY }
      return
    }
    this.drawLinker = this.draw.link(intf)
  }

  onMouseMove(e) {
    if (e.button != 0) {
      return
    }
    if (this.drawLinker) {
      this.onMouseMoveDrawLinker(e)
    } else if (this.mousedownPos) {
      this.onMouseMoveNode(e)
    }
  }
  onMouseMoveDrawLinker(e) {
    let pos = { x: e.offsetX, y: e.offsetY }
    this.drawLinker.lineTo(pos)
  }
  onMouseMoveNode(e) {
    this.currentNode.move(
      e.offsetX - this.mousedownPos.x,
      e.offsetY - this.mousedownPos.y
    )
    this.mousedownPos.x = e.offsetX
    this.mousedownPos.y = e.offsetY
  }
  onMouseUp(e) {
    if (e.button != 0) {
      return
    }
    if (this.mousedownPos) {
      this.mousedownPos = null
      return
    }

    let node = this.getMouseNode(e.offsetX, e.offsetY)
    if (this.drawLinker) {
      this.onMouseupDrawLinker(node)
    } else {
      this.onMouseupFocus(node)
    }
  }
  onMouseupFocus(node) {
    if (node && node != this.currentNode) {
      if (this.currentNode) {
        this.currentNode.focus(false)
      }
      this.currentNode = node
      node.focus(true)
    }
  }
  onMouseupDrawLinker(node) {
    if (!node || node == this.currentNode) {
      this.drawLinker.remove()
      this.drawLinker = null
      return
    }

    let pos = node.docker()
    this.drawLinker.lineTo(pos)
    this.allLinks.push(this.drawLinker)
    this.currentNode.addLinkFrom(this.drawLinker)
    node.addLinkto(this.drawLinker)
    this.drawLinker = null
  }

  getMouseNode(x, y) {
    // console.log("getMouseNode(): ", x, y)
    for (let i = this.allNodes.length - 1; i >= 0; i--) {
      if (this.allNodes[i].inside(x, y)) {
        return this.allNodes[i]
      }
    }
    return null
  }
}

四、总结

  • svg是矢量图形一种标准，可以在放大缩小时保持清晰，占用存储空间小，可无缝集成到网页中。
  • svg.js是一个轻量级的svg操作库，快速，易用，可扩展。
  • 现有的js图形库都是某方面的解决方案（如流程图库，如数据报表库等，如D3.js），不但笨重，如果对图形库有很多定制需求，这些库都无法完全满足需求。
  • 基于svg.js可以方便的扩展成自己的图形库，形成功能强大，短小精干的自定义实现库。
