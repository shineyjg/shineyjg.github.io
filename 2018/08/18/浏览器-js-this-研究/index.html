浏览器 js this 研究

2018-08-18
  • 前端
  • this 
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>this</title>
</head>
<body>
  <script>
    console.log("顶层的this是window")
    console.log('this === window', this === window)
    console.log('this', this)


    console.log('全局变量挂在window中')
    hello = 'hello'
    console.log('window.hello', window.hello)
    var world = 'world'
    console.log('window.hello', window.world)


    console.log('全局普通函数this为window')
    function test1(){
      this.name = 'hello'
      console.log('this === window', this === window)
    }
    test1()


    //测试strict时，需要把下面注释打开
    // "use strict";
    console.log('严格模式全局普通函数this为undefine')
    function test2(){
      this.name = 'hello'
    }
    test2()


    console.log('全局普通函数被new时this为相应实例')
    function test3(){
      this.name = 'hello'
    }
    var a = new test3()
    console.log(a)
    a.name = 'world'
    console.log(a)
    var b = new test3()
    console.log(b)


    console.log('全局匿名函数的this为window')
    var test4 = function(){
      this.name = 'hello'
      console.log('this === window', this === window)
    }
    test4()


    console.log('object中匿名函数this为object')
    var obj1 = {
      test: function(){
        console.log('this === obj', this === obj)
      }
    }
    obj1.test()


    console.log('object中匿名函数被全局调用时this为window')
    var obj2 = {
      test: function(){
        console.log('this === window', this === window)
      }
    }
    var f = obj2.test
    f()


    console.log('object中匿名函数中返回的匿名函数this要看绑定关系，没有特殊绑定时为window')
    var obj3 = {
      f: function(){
        return function(){
          console.log('this === window', this === window)
        }
      }
    }
    obj3.f()()


    console.log('object中匿名函数中返回的匿名函数this要看绑定关系，绑定到obj5中时为相应obj5')
    var obj4 = {
      f: function(){
        return function(){
          console.log(this)
        }
      }
    }
    var obj5 = {
      name: 'obj5',
      g: obj4.f()
    }
    obj5.g()


    console.log('object中匿名函数中返回的匿名函数this要看绑定关系，下面例子中并没有绑定为obj6，所以还是window')
    var obj6 = {
      f: function(){
        return function(){
          console.log('this === window', this === window)
        }
      }
    }
    var obj7 = {
      name: 'obj7',
      g: function(){
        obj6.f()()
      }
    }
    obj7.g()


    console.log('箭头函数的this为定义时的词法作用域，既外层调用者，或者理解为祖父（没有祖父时是父）')

    console.log('文件中箭头函数this为window')
    var test5 = ()=>{
      this.name = 'hello'
      console.log('this === window', this === window)
    }
    test5()


    console.log('object中箭头函数this为window')
    var obj8 = {
      test: ()=>{
        console.log('this === obj', this === obj)
        console.log('this === window', this === window)
      }
    }
    obj8.test()


    console.log('object中匿名成员函数中箭头函数中this为object')
    var obj9 = {
      f: function(){
        (()=>{
          console.log('this === obj', this === obj)
        })()
      }
    }
    obj9.f()


    console.log('object中匿名成员函数中返回的箭头函数中this为object')
    var obj10 = {
      f: function(){
        return ()=>{
          console.log('this === obj', this === obj)
        }
      }
    }
    obj10.f()()


    console.log('object中匿名成员函数中的箭头函数中this为window')
    var obj11 = {
      f: ()=>{
        (()=>{
          console.log('this === window', this === window)
        })()
      }
    }
    obj11.f()


    console.log('object中匿名成员函数中返回的箭头函数中this为window')
    var obj12 = {
      f: ()=>{
        return ()=>{
          console.log('this === window', this === window)
        }
      }
    }
    obj12.f()()



    console.log('call,apply改变调用时this')
    function test6(param1, param2){
      console.log(this.name, param1, param2)
    }
    var obj13 = {
      name: 'hello'
    }
    test6.call(obj13,'world', 'haha')
    test6.apply(obj13, ['world','haha'])


    console.log('call,apply无法改变箭头函数的this')
    var test7 = ()=>{
      console.log(this.name)
    }
    test7.call({name: 'hello'})
  </script>
</body>
